# --- Step 1: Import Libraries ---
import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns
# --- Step 2: Load Dataset ---
heart_data = pd.read_csv("heart.csv") # make sure heart.csv is in your working folder
# --- Step 3: Basic Info & Summary ---
print("Dataset Info:")
print(heart_data.info())
print("\nSummary Statistics:")
print(heart_data.describe())
print("\nMissing Values:")
print(heart_data.isnull().sum())
# --- Step 4: Target Variable Distribution ---
plt.figure(figsize=(6,4))
sns.countplot(x="output", data=heart_data, palette="viridis")
plt.title("Distribution of Heart Disease (0 = No, 1 = Yes)")
plt.show()
# --- Step 5: Correlation Heatmap ---
plt.figure(figsize=(12,6))
corr_matrix = heart_data.corr()
sns.heatmap(corr_matrix, annot=True, cmap="coolwarm", fmt=".2f")
plt.title("Correlation Heatmap of Features")
plt.show()49
# --- Step 6: Boxplots (Feature vs Target) ---
features = [col for col in heart_data.columns if col != "output"]
for feature in features:
plt.figure(figsize=(6,4))
sns.boxplot(x="output", y=feature, data=heart_data, palette="Set2")
plt.title(f"{feature} vs Heart Disease (Output)")
plt.show()
# --- Step 7: Histograms for Feature Distributions ---
heart_data.hist(bins=15, figsize=(15,10), layout=(4,4), color="skyblue", edgecolor="black")
plt.suptitle("Feature Distributions", fontsize=16)
plt.tight_layout()
plt.show()
# --- Step 8: Pairplot for Selected Features ---
selected = ["age", "chol", "thalachh", "oldpeak", "output"]
sns.pairplot(heart_data[selected], hue="output", palette="husl")
plt.show()
# --- Step 9: Outlier Detection (IQR method) ---
Q1 = heart_data.quantile(0.25)
Q3 = heart_data.quantile(0.75)
IQR = Q3 - Q1
outliers = ((heart_data < (Q1 - 1.5 * IQR)) | (heart_data > (Q3 + 1.5 * IQR))).sum()
print("\nNumber of Outliers per Feature:")
print(outliers)
